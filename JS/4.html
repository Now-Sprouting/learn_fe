# 手写bind方法

    Function.prototype.mybind = function () {
        // 1.将参数转换成数组
        const args = Array.prototype.slice.call(arguments)
        // 2.获取mybind传递过来的对象作为新的this
        const t = args.shift()
        // 3.self === fn
        const self = this
        return function () {
            self.apply(t, args)
        }
    }
    function fn(a, b) {
        console.log(this);
        console.log(a, b);
    }
    fn1 = fn.mybind({ a: 100 }, 1, 2)
    fn1()
